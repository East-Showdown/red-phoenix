country_event = {
    id = rus.1
    title = rus.1.t
    desc = rus.1.d
	fire_only_once = yes
	picture = GFX_report_event_SOV_leningrad_concert
    trigger = {
		tag = SOV
		has_civil_war_in_russia = yes
		date > 1994.06.17
	}
    option = {
        name = rus.1.a
		ARH = {
			set_state_owner = 214
			set_province_controller = 3338
			declare_war_on = {
				target = SOV
				type = annex_everything
			}
			create_faction = LIBERALS_NOVAYA_ROSSIYA
			load_oob = ARH_units
		}
    }
	option = {
		trigger = {
			is_ai = no
		}
        name = rus.1.b
		ARH = {
			set_state_owner = 214
			set_province_controller = 3338
			declare_war_on = {
				target = SOV
				type = annex_everything
			}
			create_faction = LIBERALS_NOVAYA_ROSSIYA
			load_oob = ARH_units
		}
		ARH = {
			change_tag_from = SOV
		}
    }
	immediate = {
		hidden_effect = {
			ARH = {
				add_ideas = RCW_russian_civil_war
			}
		}
	}
}
country_event = {
    id = rus.2
    title = rus.2.t
    desc = rus.2.d
	fire_only_once = yes
	picture = GFX_report_event_SOV_leningrad_concert
    trigger = {
		tag = ARH
		ARH = {
			controls_state = 214
		}
	}
	mean_time_to_happen = {
		days = 5
	}
    option = {
        name = rus.2.a
		KRR = {
			set_state_owner = 234
			set_province_controller = 3734
			declare_war_on = {
				target = SOV
				type = annex_everything
			}
			load_oob = KRR_units
		}
		ARH = {
			add_to_faction = KRR
		}
		if = {
			limit = { 
				has_dlc = "Together for Victory" 
			}
			ARH = {
				set_autonomy = {
					target = KRR
					autonomy_state = autonomy_collaboration_government
				}
			}

		}
			else = {
				ARH = {
				puppet = KRR
				}
			}
    }
	option = {
		trigger = {
			is_ai = no
		}
        name = rus.2.b
		KRR = {
			set_state_owner = 234
			set_province_controller = 3734
			declare_war_on = {
				target = SOV
				type = annex_everything
			}
			load_oob = KRR_units
		}
		ARH = {
			add_to_faction = KRR
		}
		if = {
			limit = { 
				has_dlc = "Together for Victory" 
			}
			ARH = {
				set_autonomy = {
					target = KRR
					autonomy_state =  autonomy_collaboration_government
				}
			}

		}
			else = {
				ARH = {
				puppet = KRR
				}
			}
		KRR = {
			change_tag_from = ARH
		}
    }
	immediate = {
		hidden_effect = {
			KRR = {
				add_ideas = RCW_russian_civil_war
			}
		}
	}
}
